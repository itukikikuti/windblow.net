こちらのコードを見てください。
``` C++
#include "XLibrary11.h";
using namespace XLibrary11;

int MAIN()
{
    Camera camera;
    
    Sprite sprite(L"temp.png");
    
    while (App::Refresh())
    {
        camera.Update();
        
        sprite.Draw();
    }
    
    return 0;
}
```
このコードを実行すると画像が１つ描画されます。
