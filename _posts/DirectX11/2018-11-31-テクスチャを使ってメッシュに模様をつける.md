IWICImagingFactoryクラスで画像の読み込みをするのでGraphicsクラスに追加します。

``` cpp
#pragma once
#include <d3d11.h>
#include <wincodec.h> // ←追加
#include <wrl.h>
#pragma comment(lib, "d3d11.lib")
#pragma comment(lib, "windowscodecs.lib") // ←追加

class Graphics
{
public:
    static void Initialize();
    static ID3D11Device& GetDevice();
    static ID3D11DeviceContext& GetContext();
    static IDXGISwapChain& GetSwapChain();
    static IWICImagingFactory& GetImageFactory(); // ←追加
    static void Update();

private:
    static Microsoft::WRL::ComPtr<ID3D11Device> device;
    static Microsoft::WRL::ComPtr<ID3D11DeviceContext> context;
    static Microsoft::WRL::ComPtr<IDXGISwapChain> swapChain;
    static Microsoft::WRL::ComPtr<IWICImagingFactory> imageFactory; // ←追加
};
```

``` cpp
...
Microsoft::WRL::ComPtr<IDXGISwapChain> Graphics::swapChain;
Microsoft::WRL::ComPtr<IWICImagingFactory> Graphics::imageFactory; // ←追加

void Graphics::Initialize()
{
    ...

    // ↓追加
    CoCreateInstance(CLSID_WICImagingFactory, nullptr, CLSCTX_INPROC_SERVER, IID_IWICImagingFactory, reinterpret_cast<void**>(imageFactory.GetAddressOf()));
}

// ↓追加
IWICImagingFactory& Graphics::GetImageFactory()
{
    return *imageFactory.Get();
}
```
